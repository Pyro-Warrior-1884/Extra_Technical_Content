#include <iostream>

using namespace std;

int main(){
    int n,i,j,target,x;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    cin>>target;
    x=0;
    for (i=0;i<n;i++){
        for (j=i+1;j<n;j++){
            if (a[i]+a[j]==target){
                cout<<"YES"<<endl<<"{"<<i<<","<<j<<"}";
                x=1;
            }
        }
    }
    if (x==0){
        cout<<"NO";
    }
    return 0;
}
