#include <iostream>

using namespace std;

int main(){
    int m,i,j,temp,start,end,z,k,c,x;
    cin>>m;
    int a[m][2];
    int b[m][2];
    int d[m][2];
    for (i=0;i<m;i++){
        for (j=0;j<2;j++){
            cin>>a[i][j];
        }
    }
    for (i=0;i<m-1;i++){
        for (j=0;j<m-i-1;j++){
            if (a[j][0]>a[j+1][0]){
                temp=a[j][0];
                a[j][0]=a[j+1][0];
                a[j+1][0]=temp;
                
                temp=a[j][1];
                a[j][1]=a[j+1][1];
                a[j+1][1]=temp;
            }
            
            else if (a[j][0]==a[j+1][0]){
                if (a[j][1]>a[j+1][1]){
                    temp=a[j][1];
                    a[j][1]=a[j+1][1];
                    a[j+1][1]=temp;
                }
            }
        }
    }
    z=0;
    for (i=0;i<m;i++){
        start=a[i][0];
        end=a[i][1];
        for (j=i+1;j<m;j++){
            if (a[i][0]<=a[j][0] && a[j][0]<=a[i][1] && a[i][1]<=a[j][1]){
                end=a[j][1];
            }
        }
        b[z][0]=start;
        b[z][1]=end;
        z++;
    }
    for (i=1;i<z;i++){
        if (b[i][0]>b[i-1][0] && b[i][0]<b[i-1][1]){
            b[i][0]=b[i-1][0];
        }
    }
    
    x=0;
    for (i=0;i<z;i++){
        c=0;
        for (j=i+1;j<z;j++){
            if (b[i][0]==b[j][0] && b[i][1]==b[j][1]){
                c++;
                if (c==1){
                    d[x][0]=b[i][0];
                    d[x][1]=b[i][1];
                    x++;
                }
            }
            
        }
    }
    
    for (i=0;i<x;i++){
        for (j=0;j<2;j++){
            cout<<d[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
