#include <iostream>

using namespace std;

int main(){
    int i,n,sw,temp,ind,j;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    
    ind=0;
    sw=0;
    
    for (i=0;i<n-1;i++){
        if (a[i]<a[i+1]){
            ind=i;//1
            break;
        }
    }
    
    for (i=ind+1;i<n;i++){
        for (j=ind+1;j<n-i+1;j++){
            if (a[j]>a[j+1]){
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
        } 
    }
    
    for (i=ind+1;i<n;i++){
        if (a[ind]<a[i]){
            sw=i;
            break;
        }
    }
    
    temp=a[sw];
    a[sw]=a[ind];
    a[ind]=temp;
    
    for (i=0;i<n;i++){
        cout<<a[i]<<" ";
    }
    
    return 0;
}
