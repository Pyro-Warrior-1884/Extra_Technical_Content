#include <iostream>

using namespace std;

int main(){
    int i,n,j,sum,max;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    max=0;
    for (i=0;i<n;i++){
        sum=0;
        for (j=i;j<n;j++){
            sum=sum+a[j];
            if (sum==0){
                if (j-i+1>max){
                    max=j-i+1;
                }
            }
        }
    }
    cout<<max;
    return 0;
}
