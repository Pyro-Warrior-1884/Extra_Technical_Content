#include <iostream>

using namespace std;

int main(){
    int n,i,j,max,min,k,temp;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    k=0;
    j=0;
    max=0;
    min=0;
    while (j<n){
        if (j%2==0){
            if (a[k]>a[max]){
                max=k;
            }
            k++;
            if (k==n){
                temp=a[j];
                a[j]=a[max];
                a[max]=temp;
                
                max=n-1;
                j++;
                k=j;
            }
        }else {
            if (a[k]<a[min]){
                min=k;
            }
            k++;
            if (k==n){
                temp=a[j];
                a[j]=a[min];
                a[min]=temp;
                
                min=n-1;
                j++;
                k=j;
            }
        }
    }
    for (i=0;i<n;i++){
        cout<<a[i]<<" ";
    }
    return 0;
}
