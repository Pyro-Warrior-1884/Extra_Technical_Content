#include <iostream>

using namespace std;

int main(){
    int i,n,j,c;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    c=0;
    for (i=0;i<n;i++){
        for (j=i+1;j<n;j++){
            if (a[j]<a[i]){
                c++;
            }
        }
    }
    cout<<"Number of Inversions: "<<c;
    return 0;
}
