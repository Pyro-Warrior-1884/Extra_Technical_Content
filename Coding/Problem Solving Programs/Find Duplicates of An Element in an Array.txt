#include <iostream>

using namespace std;

int main(){
    int n,i,j,fl;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    for (i=0;i<n;i++){
        fl=0;
        for (j=0;j<i;j++){
            if (a[i]==a[j]){
                fl++;
            }
        }
        if (fl==1){
            cout<<a[i];
        }
        
    }
    return 0;
}
