#include <iostream>

using namespace std;

int main(){
    int n,i,j,k,m,temp;
    cin>>n;
    int a[n],s[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    for (i=0;i<n;i++){
        s[i]=1;
    }
    for (i=0;i<n-1;i++){
        for (j=0;j<n-i-1;j++){
            if (a[j]>a[j+1]){
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
        }
    }
    for (i=0;i<n;i++){
        k=a[i]+1;
        for (j=0;j<n;j++){
            if (a[j]==k){
                s[i]++;
                k++;
            }
        }
    }
    
    m=0;
    for (i=0;i<n;i++){
        if (s[i]>m){
            m=s[i];
        }
    }
    cout<<m;
    return 0;
}
