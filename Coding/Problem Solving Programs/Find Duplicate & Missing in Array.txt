#include <iostream>

using namespace std;

int main(){
    int n,i,j,fl,twice,miss,c;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    for (i=0;i<n;i++){
        fl=0;
        for (j=0;j<i;j++){
            if (a[i]==a[j]){
                fl++;
            }
        }
        if (fl==1){
            twice=a[i];
        }
        
    }
    for (i=1;i<n+1;i++){
        c=0;
        for (j=0;j<n;j++){
            if (a[j]==i){
                c++;
            }
        }
        if (c==0){
            miss=i;
        }
    }
    cout<<twice<<" "<<miss;
    return 0;
}
