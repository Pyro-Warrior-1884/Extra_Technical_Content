
#include <iostream>

using namespace std;

int main(){
    int r,c,i,res,j;
    cin>>r;
    
    r=r-1;
    c=r;
    res=1;
    for (j=0;j<=c;j++){
        for (i=0;i<j;i++){
            res=res*(r-i);
            res=res/(i+1);
        }
        cout<<res<<" ";
        res=1;
    }
    
    return 0;
}
