#include <iostream>

using namespace std;

int main(){
    int i,n,mid,cs,cl;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    mid=1;
    cs=0;
    cl=0;
    while(mid<n-1){
        for (i=0;i<mid;i++){
            if (a[i]<a[mid]){
                cs++;
            }
        }
        for (i=mid+1;i<n;i++){
            if (a[i]>a[mid]){
                cl++;
            }
        }
        if ((cs+cl)==(n-1)){
            cout<<a[mid];
            break;
        }
        mid++;
        cs=0;
        cl=0;
    }
    return 0;
}
