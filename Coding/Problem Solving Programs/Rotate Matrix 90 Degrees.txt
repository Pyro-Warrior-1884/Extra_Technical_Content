#include <iostream>

using namespace std;

int main(){
    int n,i,j;
    cin>>n;
    int m[n][n],r[n][n];
    int rev[n];
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            cin>>m[i][j];
        }
    }    
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            r[i][j]=m[j][i];
        }
    }
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            rev[j]=r[i][j];
        }
        for (j=0;j<n;j++){
            r[i][j]=rev[(n-1)-j];
        }
    }
    for (i=0;i<n;i++){
        for (j=0;j<n;j++){
            cout<<r[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
