#include <iostream>

using namespace std;

int main(){
    int i,a,b,c,n;
    cin>>n;
    int arr[n];
    for (i=0;i<n;i++){
        cin>>arr[i];
    }
    
    a=0;
    b=0;
    c=0;
    for (i=0;i<n;i++){
        if (arr[i]==0){
            a++;
        } else if (arr[i]==1){
            b++;
        } else {
            c++;
        }
    }
    
    for (i=0;i<n;i++){
        if (i<a){
            arr[i]=0;
        } else if (i>=a && i<(a+b)){
            arr[i]=1;
        }else {
            arr[i]=2;
        }
    }
    
    
    for (i=0;i<n;i++){
        cout<<arr[i]<<", ";
    }
    
    return 0;
}

#include <iostream>

using namespace std;

int main(){
    int i,n,low,mid,high,temp;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    
    low=0;
    high=n-1;
    mid=0;
    
    while (mid<=high){
        if (a[mid]==0){
            temp=a[mid];
            a[mid]=a[low];
            a[low]=temp;
            low++;
            mid++;
        } else if (a[mid]==2){
            temp=a[mid];
            a[mid]=a[high];
            a[high]=temp;
            high--;
              
        } else {
            mid++;
        }
    }
    
    for (i=0;i<n;i++){
        cout<<a[i];
    }
    
    
    return 0;
}
