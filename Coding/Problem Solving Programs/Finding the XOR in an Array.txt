#include <iostream>

using namespace std;

int main(){
    int n,i,sum,count,j,target;
    cin>>n;
    int a[n];
    for (i=0;i<n;i++){
        cin>>a[i];
    }
    cin>>target;
    for (i=0;i<n;i++){
        sum=0;
        for (j=i;j<n;j++){
            sum=sum^a[j];
            if (sum==target){
                count++;
            }
        }
    }
    cout<<count;
    
    return 0;
}
